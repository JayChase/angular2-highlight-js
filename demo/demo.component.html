Raw pre code without highlighting

            
                export class HighlightJsDirective {
                    @Input() useBr: boolean;

                    constructor(el: ElementRef) {
                        if (this.useBr) {
                            hljs.configure({ useBR: true });
                        }
                        
                        hljs.highlightBlock(el.nativeElement);
                    }
                }
