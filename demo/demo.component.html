               export class HighlightJsDirective implements AfterContentInit {
                    @Input() useBr: boolean;

                    constructor(private elementRef: ElementRef, renderer: Renderer) {

                    }

                    ngAfterContentInit() {
                        if (this.useBr) {
                            hljs.configure({ useBR: true });
                        }

                        hljs.highlightBlock(this.elementRef.nativeElement);
                    }
                }
